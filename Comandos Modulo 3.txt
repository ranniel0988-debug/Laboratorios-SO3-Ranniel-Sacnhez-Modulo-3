#!/bin/bash

USUARIO=$(whoami)

FECHA=$(date +"%d-%m-%Y_%H.%M")
NOMBRE_ARCHIVO="backup_${USUARIO}_${FECHA}.tar.gz"

tar -pczf "$NOMBRE_ARCHIVO" "/home/$USUARIO"

echo "Backup creado exitosamente: $NOMBRE_ARCHIVO"

#dar permisos de ejecucion
sudo chmod +x backup.sh
# ejecutar el script
./backup.sh
